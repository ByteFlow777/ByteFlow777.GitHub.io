      服务端代码 #

      服务端（被动）接受连接请求的套接字创建过程如下：

      1. 调用 socket 函数创建套接字。
      2. 调用 bind 函数分配 IP 地址和端口号。
      3. 调用 listen 函数转为可接受请求状态。
      4. 调用 accept 函数处理连接请求。
      5. 调用 read 和 write 函数进行数据交换。
      6. 调用 close 函数断开连接。
      #include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <unistd.h>

#include <arpa/inet.h>
#include <sys/socket.h>

void panic(char *err_message) {
    fputs(err_message, stderr);
    fputc('\n', stderr);
    exit(EXIT_FAILURE);
}

int main(int argc, char *argv[]) {
    if (argc != 2) {
        panic("Usage: <program_name> <port>");
    }

    // server port
    int port = atoi(argv[1]);

    // PF_INET: IPv4 Internet protocols
    // SOCK_STREAM: TCP
    // 0: automatically choose
    int server_sock_fd = socket(PF_INET, SOCK_STREAM, 0);
    if (server_sock_fd == -1) {
        panic("socket() error");
    }

    struct sockaddr_in server_addr;
    memset(&server_addr, 0, sizeof(server_addr));
    server_addr.sin_family = AF_INET;
    server_addr.sin_addr.s_addr = htonl(INADDR_ANY);
    server_addr.sin_port = htons(port);

    if (bind(server_sock_fd, (struct sockaddr *)&server_addr,
             sizeof(server_addr)) == -1) {
        panic("bind() error");
    }

    if (listen(server_sock_fd, 5) == -1) {
        panic("listen() error");
    }

    struct sockaddr_in client_addr;
    socklen_t client_addr_size = sizeof(client_addr);
    int client_sock_fd = accept(server_sock_fd, (struct sockaddr *)&client_addr,
                                &client_addr_size);
    if (client_sock_fd == -1) {
        panic("accept() error");
    }

    char message[] = "Hello World";
    write(client_sock_fd, message, sizeof(message));

    close(client_sock_fd);
    close(server_sock_fd);

    return 0;
}

      客户端代码 #

      客户端（主动）连接请求的套接字创建过程如下：

      1. 调用 socket 函数创建套接字。
      2. 调用 connect 函数与服务器建立连接。
      3. 调用 read 和 write 函数进行数据交换。
      4. 调用 close 函数断开连接。
      #include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <unistd.h>

#include <arpa/inet.h>
#include <sys/socket.h>

void panic(char *err_message) {
    fputs(err_message, stderr);
    fputc('\n', stderr);
    exit(EXIT_FAILURE);
}

int main(int argc, char *argv[]) {
    if (argc != 3) {
        panic("Usage: <program_name> <IP> <port>");
    }

    int sock_fd = socket(PF_INET, SOCK_STREAM, 0);
    if (sock_fd == -1) {
        panic("socket() error");
    }

    struct sockaddr_in server_addr;
    memset(&server_addr, 0, sizeof(server_addr));
    server_addr.sin_family = AF_INET;
    server_addr.sin_addr.s_addr = inet_addr(argv[1]);
    server_addr.sin_port = htons(atoi(argv[2]));

    if (connect(sock_fd, (struct sockaddr *)&server_addr,
                sizeof(server_addr)) == -1) {
        panic("connect() error");
    }

    char message[128];
    ssize_t read_size = read(sock_fd, message, sizeof(message) - 1);
    if (read_size == -1) {
        panic("read() error");
    }
    message[read_size] = '\0';

    printf("%s\n", message);

    close(sock_fd);
    return 0;
}

      改进的服务端代码 #

      上边的服务端代码只能处理一次客户端的连接，我们可以对其进行改进，当处理完某个客户端请求后，可以重新调用 accept 函数：

      while (1) {
    struct sockaddr_in client_addr;
    socklen_t client_addr_size = sizeof(client_addr);

    int client_sock_fd = accept(server_sock_fd, (struct sockaddr *) &client_addr,
                                &client_addr_size);
    if (client_sock_fd == -1) {
        panic("accept() error");
    }

    char message[] = "Hello World";
    write(client_sock_fd, message, sizeof(message));

    close(client_sock_fd);
}
